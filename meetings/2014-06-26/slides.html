Rekursja ogonowa

user=> (defn print-down-from [x]
  #_=>   (when (pos? x)
  #_=>     (println x)
  #_=>     (recur (dec x))))

clojure.test

(ns your-namespace
  (:require clojure.test))

(testing "Arithmetic"
  (testing "with positive integers"
    (is (= 7 (+ 3 4))))
  (testing "with negative integers"
    (is (= -1 (+ 3 -4)))))
